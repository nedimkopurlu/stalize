    const closedStats = useMemo(() => {

        const pct = p.realized_pnl !== null && p.entry_price > 0 && p.exit_price !== null
          ? ((p.exit_price - p.entry_price) / p.entry_price) * 100
          : null;
        return pct !== null ? sum + pct : sum;
      }, 0) / closedPositions.filter(p =>

      const plannedCount = closedPositions.filter(

      ).length;

      return { count: closedPositions.length, avgPnlPct, plannedPct };
