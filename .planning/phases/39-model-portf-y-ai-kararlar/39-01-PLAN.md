    function ModelPortfolioHistory() {
      const [history, setHistory] = useState<ModelPortfolioHistoryResponse | null>(null);
      const [loading, setLoading] = useState(true);

      useEffect(() => {
        void (async () => {
          try {
            const res = await api.getModelPortfolioHistory(8);
            setHistory(res);
          } catch {
            /* non-critical, fail silently */
          } finally {
            setLoading(false);
          }
        })();
      }, []);

      const weeks = history?.weeks ?? [];

      return (
        <section className={styles.historySection}>
          <h2 className={styles.historySectionTitle}>Geçmiş Portföyler</h2>
          {loading && <p className={styles.historyLoading}>Yükleniyor...</p>}
          {!loading && weeks.length === 0 && (
            <p className={styles.historyEmpty}>Henüz geçmiş portföy kaydı yok.</p>
          )}
          {!loading && weeks.length > 0 && (
            <div className={styles.historyList}>
              {weeks.map((w) => {
                const returnPct = w.portfolio_return_pct;
                const isGreen = (returnPct ?? 0) >= 0;
                return (
                  <div key={w.id} className={styles.historyRow}>
                    <span className={styles.historyWeek}>
                      {new Date(w.week_start).toLocaleDateString('tr-TR', { day: 'numeric', month: 'short' })}
                      {' – '}
                      {new Date(w.week_end).toLocaleDateString('tr-TR', { day: 'numeric', month: 'short', year: 'numeric' })}
                    </span>
                    <span
                      className={styles.historyReturn}
                      style={{ color: isGreen ? 'var(--accent-green, #10b981)' : 'var(--accent-red, #ef4444)' }}
                    >
                      {formatPct(returnPct)}
                    </span>
                    <span className={styles.historyBist}>{formatPct(w.benchmark_return_pct)}</span>
                    {w.review_summary && (
                      <span className={styles.historyReview}>{w.review_summary}</span>
                    )}
                  </div>
                );
              })}
            </div>
          )}
        </section>
      );
    }
