    ```python
    """
    Tests for MLCA-03 — LLM analysis result caching via diskcache.
    All tests mock DeepSeekSentimentService.client; no real API calls are made.
    """
    import pytest
    import diskcache
    from unittest.mock import AsyncMock, patch, MagicMock
    from datetime import date


    def make_mock_response(impact_score: int = 30):
        """Build a mock AsyncOpenAI response object."""
        mock_response = MagicMock()
        mock_response.choices = [MagicMock()]
        mock_response.choices[0].message.content = (
            f'{{"impact_score": {impact_score}, "direction": "bullish", '
            f'"affected_symbols": ["AKBNK"], "summary": "Test", "reasoning": "Test reason"}}'
        )
        return mock_response


    @pytest.mark.asyncio
    async def test_cache_hit_skips_api(tmp_path):
        """Second analyze() call with same args does not call DeepSeek API."""
        import app.services.llm_sentiment as ls

        test_cache = diskcache.Cache(str(tmp_path / "llm_test"))
        with patch.object(ls, "_llm_cache", test_cache):
            service = ls.DeepSeekSentimentService.__new__(ls.DeepSeekSentimentService)
            service.api_key = "test_key"
            service.model = "deepseek-chat"
            service.client = MagicMock()
            service.client.chat = MagicMock()
            service.client.chat.completions = MagicMock()
            service.client.chat.completions.create = AsyncMock(return_value=make_mock_response())

            kwargs = dict(title="KAP Test Haberi", summary="", source="KAP", symbol="AKBNK")
            await service.analyze(**kwargs)   # first call — cache miss, API called
            await service.analyze(**kwargs)   # second call — cache hit, API NOT called

            assert service.client.chat.completions.create.call_count == 1


    @pytest.mark.asyncio
    async def test_cache_key_format(tmp_path):
        """Cache key contains ticker + today's date + hash of title."""
        import app.services.llm_sentiment as ls

        test_cache = diskcache.Cache(str(tmp_path / "llm_key"))
        stored_keys = []
        original_set = test_cache.set

        def capturing_set(key, value, **kwargs):
            stored_keys.append(key)
            return original_set(key, value, **kwargs)

        test_cache.set = capturing_set

        with patch.object(ls, "_llm_cache", test_cache):
            service = ls.DeepSeekSentimentService.__new__(ls.DeepSeekSentimentService)
            service.api_key = "test_key"
            service.model = "deepseek-chat"
            service.client = MagicMock()
            service.client.chat.completions.create = AsyncMock(return_value=make_mock_response())

            title = "Test Haber Başlığı"
            await service.analyze(title=title, symbol="AKBNK")

        assert len(stored_keys) == 1
        key = stored_keys[0]
        today = date.today().isoformat()
        assert key.startswith("analysis:AKBNK:")
        assert today in key
        assert str(hash(title)) in key


    @pytest.mark.asyncio
    async def test_cache_expiry(tmp_path):
        """Cache.set is called with expire=1800 (30 minutes)."""
        import app.services.llm_sentiment as ls

        test_cache = diskcache.Cache(str(tmp_path / "llm_expiry"))
        expire_values = []
        original_set = test_cache.set

        def capturing_set(key, value, **kwargs):
            expire_values.append(kwargs.get("expire"))
            return original_set(key, value, **kwargs)

        test_cache.set = capturing_set

        with patch.object(ls, "_llm_cache", test_cache):
            service = ls.DeepSeekSentimentService.__new__(ls.DeepSeekSentimentService)
            service.api_key = "test_key"
            service.model = "deepseek-chat"
            service.client = MagicMock()
            service.client.chat.completions.create = AsyncMock(return_value=make_mock_response())

            await service.analyze(title="Expire Test", symbol="GARAN")

        assert len(expire_values) == 1
        assert expire_values[0] == 1800
    ```
