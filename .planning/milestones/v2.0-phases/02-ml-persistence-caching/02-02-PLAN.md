       ```python
       def _model_path(self, symbol: str) -> str:
           return os.path.join(MODEL_DIR, f"{symbol}_xgb.ubj")

       def _scaler_path(self, symbol: str) -> str:
           return os.path.join(MODEL_DIR, f"{symbol}_scaler.joblib")

       def save_model(self, symbol: str, model, scaler) -> None:
           """Save XGBRegressor and StandardScaler to disk as paired artifacts."""
           os.makedirs(MODEL_DIR, exist_ok=True)
           model.save_model(self._model_path(symbol))
           joblib.dump(scaler, self._scaler_path(symbol))
           logger.info(f"Model kaydedildi: {symbol} → {MODEL_DIR}")

       def load_model(self, symbol: str):
           """Load (model, scaler) from disk. Returns (None, None) if not found."""
           mp = self._model_path(symbol)
           sp = self._scaler_path(symbol)
           if not os.path.exists(mp) or not os.path.exists(sp):
               return None, None
           model = xgb.XGBRegressor()
           model.load_model(mp)
           scaler = joblib.load(sp)
           logger.info(f"Model diskten yüklendi: {symbol}")
           return model, scaler

       def preload_all_models(self) -> int:
           """Scan MODEL_DIR for *.ubj files and load all into self._models. Returns count."""
           self._models = {}
           if not os.path.isdir(MODEL_DIR):
               logger.info("models/ dizini yok — diskte model bulunmadı")
               return 0
           for fname in os.listdir(MODEL_DIR):
               if fname.endswith("_xgb.ubj"):
                   symbol = fname[:-8]  # strip "_xgb.ubj"
                   model, scaler = self.load_model(symbol)
                   if model is not None:
                       self._models[symbol] = (model, scaler)
           count = len(self._models)
           logger.info(f"XGBoost modelleri preload: {count} hisse")
           return count

       async def retrain_and_save(self, db: AsyncSession, stock) -> None:
           """Force full retrain for stock and overwrite persisted files."""
           try:
               df = await self._fetch_historical_data(db, stock.id)
               if df.empty or len(df) < 50:
                   logger.warning(f"Retrain için yetersiz veri: {stock.symbol}")
                   return
               X, y, _ = self._prepare_features(df)
               if X is None or len(X) < 10:
                   return
               if not XGBOOST_AVAILABLE:
                   return
               scaler = StandardScaler()
               X_scaled = scaler.fit_transform(X)
               model = xgb.XGBRegressor(**self.xgb_params)
               model.fit(X_scaled, y)
               self.save_model(stock.symbol, model, scaler)
               self._models[stock.symbol] = (model, scaler)
               logger.info(f"Retrain tamamlandı: {stock.symbol}")
           except Exception as e:
               logger.error(f"Retrain hatası ({stock.symbol}): {e}")
       ```

       ```python
       async def _train_and_predict(self, X, y, current_features, symbol: str = None) -> Optional[float]:
           try:
               if not XGBOOST_AVAILABLE:
                   return float(current_features[2] * 0.5)

               # Check in-memory cache first
               if symbol and symbol in self._models:
                   model, scaler = self._models[symbol]
                   curr_scaled = scaler.transform(current_features.reshape(1, -1))
                   return float(model.predict(curr_scaled)[0])

               # Train from scratch
               scaler = StandardScaler()
               X_scaled = scaler.fit_transform(X)
               model = xgb.XGBRegressor(**self.xgb_params)
               model.fit(X_scaled, y)

               # Persist and cache in-memory
               if symbol:
                   self.save_model(symbol, model, scaler)
                   self._models[symbol] = (model, scaler)

               curr_scaled = scaler.transform(current_features.reshape(1, -1))
               return float(model.predict(curr_scaled)[0])
           except Exception as e:
               logger.error(f"Tahminleme hatası: {e}")
       ```
