    ```tsx
    'use client';

    import React, { useEffect, useState } from 'react';
    import Sidebar from '@/components/Sidebar';
    import api, { PortfolioPosition } from '@/lib/api';
    import { formatPrice } from '@/components/StockHelpers';
    import Link from 'next/link';

    // ── PositionCard ─────────────────────────────────────────
    function PositionCard({ position }: { position: PortfolioPosition }) {
      const gainLoss = position.gain_loss_pct;
      const isPositive = gainLoss !== null && gainLoss >= 0;
      const gainColor = gainLoss === null
        ? 'var(--text-muted)'
        : isPositive ? 'var(--green-400)' : 'var(--red-400)';

      return (
        <div className="card" style={{ padding: '20px 24px', display: 'flex', flexDirection: 'column', gap: 12 }}>
          {/* Header row: symbol + gain/loss badge */}
          <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center' }}>
            <Link href={`/stocks/${position.symbol}`} style={{ textDecoration: 'none' }}>
              <span style={{ fontWeight: 700, fontSize: '1.05rem', color: 'var(--text-primary)' }}>
                {position.symbol}
              </span>
            </Link>
            {gainLoss !== null ? (
              <span style={{
                fontSize: '0.9rem',
                fontWeight: 600,
                color: gainColor,
                background: isPositive ? 'var(--green-bg)' : 'var(--red-bg)',
                border: `1px solid ${isPositive ? 'var(--green-border)' : 'var(--red-border)'}`,
                borderRadius: 6,
                padding: '2px 10px',
              }}>
                {gainLoss >= 0 ? '+' : ''}{gainLoss.toFixed(2)}%
              </span>
            ) : (
              <span style={{ fontSize: '0.85rem', color: 'var(--text-muted)' }}>—</span>
            )}
          </div>

          {/* Price row */}
          <div style={{ display: 'grid', gridTemplateColumns: '1fr 1fr', gap: 12 }}>
            <div>
              <div style={{ fontSize: '0.7rem', color: 'var(--text-muted)', marginBottom: 2 }}>Giriş Fiyatı</div>
              <div style={{ fontFamily: "'JetBrains Mono', monospace", fontSize: '0.9rem' }}>
                {formatPrice(position.entry_price)}
              </div>
            </div>
            <div>
              <div style={{ fontSize: '0.7rem', color: 'var(--text-muted)', marginBottom: 2 }}>Güncel Fiyat</div>
              <div style={{ fontFamily: "'JetBrains Mono', monospace", fontSize: '0.9rem', color: gainColor }}>
                {position.current_price !== null ? formatPrice(position.current_price) : '—'}
              </div>
            </div>
          </div>

          {/* Stop-loss / target row */}
          <div style={{ display: 'grid', gridTemplateColumns: '1fr 1fr', gap: 12, borderTop: '1px solid var(--border-primary)', paddingTop: 10 }}>
            <div>
              <div style={{ fontSize: '0.7rem', color: 'var(--text-muted)', marginBottom: 2 }}>Stop-Loss</div>
              <div style={{ fontFamily: "'JetBrains Mono', monospace", fontSize: '0.85rem', color: 'var(--red-400)' }}>
                {position.stop_loss !== null ? formatPrice(position.stop_loss) : '—'}
              </div>
            </div>
            <div>
              <div style={{ fontSize: '0.7rem', color: 'var(--text-muted)', marginBottom: 2 }}>Hedef</div>
              <div style={{ fontFamily: "'JetBrains Mono', monospace", fontSize: '0.85rem', color: 'var(--green-400)' }}>
                {position.target_price !== null ? formatPrice(position.target_price) : '—'}
              </div>
            </div>
          </div>

          {/* Rationale (optional) */}
          {position.rationale && (
            <div style={{ fontSize: '0.75rem', color: 'var(--text-muted)', borderTop: '1px solid var(--border-primary)', paddingTop: 8 }}>
              {position.rationale}
            </div>
          )}
        </div>
      );
    }

    // ── Page ─────────────────────────────────────────────────
    export default function PortfolioPage() {
      const [positions, setPositions] = useState<PortfolioPosition[]>([]);
      const [loading, setLoading] = useState(true);
      const [error, setError] = useState<string | null>(null);

      useEffect(() => {
        api.getPortfolioPositions()
          .then(setPositions)
          .catch((err: Error) => setError(err.message))
          .finally(() => setLoading(false));
      }, []);

      return (
        <div className="app-layout">
          <Sidebar />
          <main className="main-content">
            <div style={{ marginBottom: 24 }}>
              <h1 className="page-title">Model Portföy</h1>
              <p className="page-subtitle">Aktif pozisyonlar ve performans takibi</p>
            </div>

            {/* Positions section */}
            <section style={{ marginBottom: 40 }}>
              <h2 style={{ fontSize: '1rem', fontWeight: 600, color: 'var(--text-secondary)', marginBottom: 16 }}>
                Aktif Pozisyonlar
              </h2>

              {loading && (
                <div style={{ display: 'grid', gridTemplateColumns: 'repeat(auto-fill, minmax(280px, 1fr))', gap: 16 }}>
                  {Array.from({ length: 3 }).map((_, i) => (
                    <div key={i} className="card skeleton" style={{ height: 160 }} />
                  ))}
                </div>
              )}

              {!loading && error && (
                <div className="card" style={{ padding: 40, textAlign: 'center', color: 'var(--red-400)' }}>
                  Pozisyonlar yüklenemedi: {error}
                </div>
              )}

              {!loading && !error && positions.length === 0 && (
                <div className="card" style={{ padding: 60, textAlign: 'center', color: 'var(--text-muted)' }}>
                  Henüz pozisyon yok
                </div>
              )}

              {!loading && !error && positions.length > 0 && (
                <div style={{ display: 'grid', gridTemplateColumns: 'repeat(auto-fill, minmax(280px, 1fr))', gap: 16 }}>
                  {positions.map((pos) => (
                    <PositionCard key={pos.id} position={pos} />
                  ))}
                </div>
              )}
            </section>

            {/* Placeholder sections for Plan 02 */}
            <section style={{ marginBottom: 40 }}>
              <h2 style={{ fontSize: '1rem', fontWeight: 600, color: 'var(--text-secondary)', marginBottom: 16 }}>
                Performans Takvimi
              </h2>
              <div className="card" style={{ padding: 40, textAlign: 'center', color: 'var(--text-muted)', minHeight: 200, display: 'flex', alignItems: 'center', justifyContent: 'center' }}>
                <span>Takvim bileşeni yükleniyor (Plan 02)</span>
              </div>
            </section>

            <section>
              <h2 style={{ fontSize: '1rem', fontWeight: 600, color: 'var(--text-secondary)', marginBottom: 16 }}>
                BIST100 Karşılaştırma
              </h2>
              <div className="card" style={{ padding: 40, textAlign: 'center', color: 'var(--text-muted)', minHeight: 200, display: 'flex', alignItems: 'center', justifyContent: 'center' }}>
                <span>Grafik Phase 15'te gerçek veriyle doldurulacak</span>
              </div>
            </section>
          </main>
        </div>
      );
    }
    ```
