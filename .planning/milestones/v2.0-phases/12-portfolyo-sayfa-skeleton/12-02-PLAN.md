    Implementation logic:
    1. Build a `Map<string, number | null>` from snapshots: date → daily_pnl_pct
    2. Determine the 12-month range to display: from the first day of (today - 11 months) through today
    3. For each month, render a column group. Inside each group, render 4-5 week columns of 7 day cells each.
    4. Each cell is a 12×12px square div (gap: 2px between cells):
       - `daily_pnl_pct > 0` → background `var(--green-500)`, opacity scales with magnitude (min 0.4, max 1.0 at 3%+ gain)
       - `daily_pnl_pct < 0` → background `var(--red-500)`, opacity scales with magnitude
       - `daily_pnl_pct === 0` → background `rgba(100,116,139,0.2)` (market open, flat)
       - no data (null or date not in map) → background `rgba(100,116,139,0.08)` (gray, market closed or no snapshot)
       - future dates → skip rendering (empty space)
    5. Month labels (Jan, Feb, ... in Turkish: Oca, Şub, Mar, Nis, May, Haz, Tem, Ağu, Eyl, Eki, Kas, Ara) above each month column group
    6. Day-of-week labels (Pt, Sa, Ça, Pe, Cu) at left edge (optional — can skip for brevity)
    7. Tooltip on hover: show date + pnl_pct formatted as "+1.23%" or "Veri yok"

    Layout: wrap entire calendar in the `.card` div. Add `overflowX: 'auto'` so it scrolls on small screens.

    Turkish month abbreviations array to use:
    ```typescript
    const MONTHS_TR = ['Oca', 'Şub', 'Mar', 'Nis', 'May', 'Haz', 'Tem', 'Ağu', 'Eyl', 'Eki', 'Kas', 'Ara'];
    ```

    Cell color helper:
    ```typescript
    function cellColor(pnl: number | null): string {
      if (pnl === null) return 'rgba(100,116,139,0.08)';
      if (pnl === 0) return 'rgba(100,116,139,0.2)';
      const intensity = Math.min(Math.abs(pnl) / 3, 1); // 3% = full intensity
      const alpha = 0.4 + intensity * 0.6;
      return pnl > 0
        ? `rgba(34,197,94,${alpha.toFixed(2)})`
        : `rgba(239,68,68,${alpha.toFixed(2)})`;
    }
    ```

    ```typescript
    'use client';

    import React, { useEffect, useRef } from 'react';
    import { createChart, ColorType } from 'lightweight-charts';

    export default function BistComparisonChart() {
      const containerRef = useRef<HTMLDivElement>(null);

      useEffect(() => {
        if (!containerRef.current) return;

        const chart = createChart(containerRef.current, {
          width: containerRef.current.clientWidth,
          height: 300,
          layout: {
            background: { type: ColorType.Solid, color: 'transparent' },
            textColor: '#94a3b8',
          },
          grid: {
            vertLines: { color: 'rgba(148,163,184,0.08)' },
            horzLines: { color: 'rgba(148,163,184,0.08)' },
          },
          rightPriceScale: { borderColor: 'rgba(148,163,184,0.1)' },
          timeScale: { borderColor: 'rgba(148,163,184,0.1)' },
        });

        // Portfolio line (skeleton — empty data, Phase 15 fills this)
        const portfolioSeries = chart.addLineSeries({
          color: '#22d3ee',
          lineWidth: 2,
          title: 'Portföy',
        });
        portfolioSeries.setData([]);

        // BIST100 comparison line (skeleton — empty data, Phase 15 fills this)
        const bistSeries = chart.addLineSeries({
          color: '#a855f7',
          lineWidth: 2,
          title: 'BIST100',
        });
        bistSeries.setData([]);

        const handleResize = () => {
          if (containerRef.current) {
            chart.applyOptions({ width: containerRef.current.clientWidth });
          }
        };
        window.addEventListener('resize', handleResize);

        return () => {
          window.removeEventListener('resize', handleResize);
          chart.remove();
        };
      }, []);

      return (
        <div className="card" style={{ padding: '20px 24px' }}>
          <div style={{ display: 'flex', gap: 20, marginBottom: 16, alignItems: 'center' }}>
            <span style={{ fontSize: '0.9rem', fontWeight: 600, color: 'var(--text-primary)' }}>
              Kümülatif Getiri
            </span>
            <div style={{ display: 'flex', gap: 12, marginLeft: 'auto' }}>
              <span style={{ fontSize: '0.75rem', color: 'var(--text-muted)', display: 'flex', alignItems: 'center', gap: 4 }}>
                <span style={{ width: 12, height: 2, background: '#22d3ee', display: 'inline-block', borderRadius: 1 }} />
                Portföy
              </span>
              <span style={{ fontSize: '0.75rem', color: 'var(--text-muted)', display: 'flex', alignItems: 'center', gap: 4 }}>
                <span style={{ width: 12, height: 2, background: '#a855f7', display: 'inline-block', borderRadius: 1 }} />
                BIST100
              </span>
            </div>
          </div>
          <div style={{ color: 'var(--text-muted)', fontSize: '0.8rem', marginBottom: 8 }}>
            Gerçek veri Phase 15'te eklenecek
          </div>
          <div ref={containerRef} style={{ width: '100%', minHeight: 300 }} />
        </div>
      );
    }
    ```
